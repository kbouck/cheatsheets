```bash
cat -b             # numberize non-blank lines, reset after blank line
cat -n             # numberize all lines

cat -v             # display non-printing chars 
cat -t             # display non-printing chars (and tabs as ^I)
cat -e             # display non-printing chars (and end-of-line as $)

cat -s             # squeeze adjacent empty lines 
cat -u             # disable output buffering
```
